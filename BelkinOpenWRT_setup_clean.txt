Blocking LLDP packets
```
# Block incoming LLDP on WAN
ebtables -A INPUT -i wan -d 01:80:c2:00:00:0e -j DROP

# Block outgoing LLDP on WAN  
ebtables -A OUTPUT -o wan -d 01:80:c2:00:00:0e -j DROP

# Block forwarding LLDP through WAN
ebtables -A FORWARD -i wan -d 01:80:c2:00:00:0e -j DROP
ebtables -A FORWARD -o wan -d 01:80:c2:00:00:0e -j DROP
	
# Make it persistent
nano /etc/rc.local
	
# Write before 'exit 0'
ebtables -A INPUT -i wan -d 01:80:c2:00:00:0e -j DROP
ebtables -A OUTPUT -o wan -d 01:80:c2:00:00:0e -j DROP
ebtables -A FORWARD -i wan -d 01:80:c2:00:00:0e -j DROP
ebtables -A FORWARD -o wan -d 01:80:c2:00:00:0e -j DROP

# Verify
ebtables -L

# ADDITIONAL KNOWLEDGE:
ebtables -F # to clear all current rules in ebtables
```


Set TTL and HL to 128 (standard to Windows)
```
# Open firewall config file
nano /etc/config/firewall

# Add new block BEFORE first "rule" block
config include
    option path '/etc/firewall.user'
    option fw4_compatible '1'

# Open firewall.user file
nano /etc/firewall.user

# Write out rules for IPv4 and for IPv6 accordingly
nft add rule inet fw4 mangle_postrouting oifname wan ip ttl set 128
nft add rule inet fw4 mangle_postrouting oifname wan ip6 hoplimit set 128
```

Applying proxy to DNS over HTTPS (DoH) using Cloudflare
```
# Set the bootstrap DNS servers for the first https-dns-proxy instance to Cloudflare’s DNS (1.1.1.1 and 1.0.0.1)
uci set https-dns-proxy.@https-dns-proxy[0].bootstrap_dns='1.1.1.1,1.0.0.1'

# Sets the DoH endpoint URL that the proxy will use to actually perform DoH lookups
uci set https-dns-proxy.@https-dns-proxy[0].resolver_url='https://cloudflare-dns.com/dns-query'

# Save the changes
uci commit https-dns-proxy

# Enables the new service to start automatically on boot
/etc/init.d/https-dns-proxy enable

# (Optional) Start it right away
/etc/init.d/https-dns-proxy start

# Verify
/etc/init.d/https-dns-proxy status
```
